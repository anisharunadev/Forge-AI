```python
#!/usr/bin/env python3
"""Regenerate docs/reference/api-catalog.md from backend/app/api/v1/**.

Ponytail: uses stdlib ast only — no libcst / parso / rope. FastAPI routers
all use the same decorator shape (@router.<method>("<path>", ...)) which
ast.parse + ast.walk handles in one pass.

Usage:
    ./scripts/gen-api-catalog.py            # rewrite the file
    ./scripts/gen-api-catalog.py --check    # exit 1 if drift
    ./scripts/gen-api-catalog.py --dry-run  # print to stdout

Output:
    docs/reference/api-catalog.md
"""
from __future__ import annotations

import argparse
import ast
import datetime as dt
import re
import sys
from pathlib import Path

REPO = Path(__file__).resolve().parent.parent
V1 = REPO / "backend" / "app" / "api" / "v1"
OUT = REPO / "docs" / "reference" / "api-catalog.md"
PREFIX = "/api/v1"  # all routes mount under this; FastAPI default in main.py


def collect_routes() -> list[dict]:
    """Walk V1 for *.py files, parse, extract @router.<method> decorators.

    Returns list of {file, method, path, docstring} sorted by file then path.
    Sub-bucket files (ideation/, architecture/, forge_phase4/) use
    `ideation.ideas.router` style includes — we record the file's relative
    path; the prefix comes from the @router.<method>("/...") arg.
    """
    routes: list[dict] = []
    for py in sorted(V1.rglob("*.py")):
        rel = py.relative_to(V1)
        if rel.name in {"__init__.py", "router.py", "_package_wiring.py"}:
            continue
        try:
            tree = ast.parse(py.read_text(encoding="utf-8"))
        except SyntaxError as e:  # pragma: no cover
            print(f"warn: cannot parse {py}: {e}", file=sys.stderr)
            continue
        for node in ast.walk(tree):
            if not isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                continue
            for dec in node.decorator_list:
                call = dec if isinstance(dec, ast.Call) else None
                if call is None:
                    continue
                func = call.func
                # Match @router.<method>("<path>") and
                # @<local>.router.<method>("<path>").
                attr = None
                if isinstance(func, ast.Attribute) and func.attr in {
                    "get", "post", "put", "patch", "delete"
                }:
                    attr = func.attr
                if attr is None:
                    continue
                if not call.args:
                    continue
                first = call.args[0]
                if not isinstance(first, ast.Constant) or not isinstance(first.value, str):
                    continue
                path = first.value
                # Docstring = first statement's Constant value, or empty.
                doc = ast.get_docstring(node) or ""
                summary = doc.splitlines()[0].strip() if doc else ""
                routes.append({
                    "file": str(rel),
                    "method": attr.upper(),
                    "path": path,
                    "summary": summary,
                })
    routes.sort(key=lambda r: (r["file"], r["path"]))
    return routes


def render(routes: list[dict], today: str) -> str:
    by_file: dict[str, list[dict]] = {}
    for r in routes:
        by_file.setdefault(r["file"], []).append(r)
    lines = [
        "# Reference: API Catalog (All Backend Routes)",
        "",
        "<!-- AUTO-GENERATED. DO NOT EDIT. Regenerate via ./scripts/gen-api-catalog.py -->",
        "",
        "> **Status:** ✅ Auto-generated",
        "> **Doc owner:** Platform team",
        f"> **Source of truth:** `backend/app/api/v1/`",
        f"> **Last regenerated:** {today}",
        f"> **Total routes:** {len(routes)}",
```

**Pre-conditions:** Phase 2 green.

**Files edited:**

**Critical decision (ponytail):** "import every model in a subprocess" (per the brief) is heavy — each model imports a `Base`, which requires a configured engine. Ponytail: **parse the `.py` files with `ast` and extract the model classes' `__tablename__` and `mapped_column(...)` calls**. This avoids booting SQLAlchemy, the app, or env vars. The brief's "subprocess" wording is one option; AST is shorter and equally correct for a doc generator.

**Exact script `scripts/gen-db-schema.py`:**

```python
#!/usr/bin/env python3
"""Regenerate docs/reference/db-schema.md from backend/app/db/models/**.

Ponytail: parses each .py with stdlib ast. Imports SQLAlchemy Base only to
resolve mapped_column() argument names — but actually we don't need Base
at all: each model file declares __tablename__ and mapped_column(...)
calls as source code, and ast can read them directly.

Usage:
    ./scripts/gen-db-schema.py            # rewrite the file
    ./scripts/gen-db-schema.py --check    # exit 1 if drift
    ./scripts/gen-db-schema.py --dry-run  # print to stdout
"""
from __future__ import annotations

import argparse
import ast
import datetime as dt
import sys
from pathlib import Path

REPO = Path(__file__).resolve().parent.parent
MODELS = REPO / "backend" / "app" / "db" / "models"
OUT = REPO / "docs" / "reference" / "db-schema.md"


def _const_str(node: ast.AST) -> str | None:
    if isinstance(node, ast.Constant) and isinstance(node.value, str):
        return node.value
    return None


def _short_type(node: ast.AST) -> str:
    """Render a mapped_column(...) type annotation as a short string."""
    if isinstance(node, ast.Name):
        return node.id
    if isinstance(node, ast.Attribute):
        return f"{_short_type(node.value)}.{node.attr}"
    if isinstance(node, ast.Subscript):
        base = _short_type(node.value)
        sl = node.slice
        if isinstance(sl, ast.Name):
            return f"{base}[{sl.id}]"
        if isinstance(sl, ast.Tuple):
            parts = [_short_type(e) for e in sl.elts]
            return f"{base}[{', '.join(parts)}]"
        if isinstance(sl, ast.Constant):
            return f"{base}[{sl.value!r}]"
        return f"{base}[...]"
    if isinstance(node, ast.Call):
        return _short_type(node.func)
    return type(node).__name__


def _kwargs_of(call: ast.Call) -> dict[str, ast.AST]:
    out: dict[str, ast.AST] = {}
    for kw in call.keywords:
        if kw.arg is not None:
            out[kw.arg] = kw.value
    return out


def collect_models() -> list[dict]:
    """Walk MODELS for *.py files, extract class definitions that look like
    SQLAlchemy models (subclass Base or have __tablename__)."""
    models: list[dict] = []
    for py in sorted(MODELS.glob("*.py")):
        if py.name == "__init__.py":
            continue
        try:
            tree = ast.parse(py.read_text(encoding="utf-8"))
        except SyntaxError as e:  # pragma: no cover
            print(f"warn: cannot parse {py}: {e}", file=sys.stderr)
            continue
        for node in ast.walk(tree):
            if not isinstance(node, ast.ClassDef):
                continue
            tablename = None
            cols: list[dict] = []
            for stmt in node.body:
                if isinstance(stmt, ast.Assign):
                    for tgt in stmt.targets:
                        if (
                            isinstance(tgt, ast.Name)
                            and tgt.id == "__tablename__"
                            and isinstance(stmt.value, ast.Constant)
                        ):
                            tablename = stmt.value.value
                if (
                    isinstance(stmt, ast.AnnAssign)
                    and isinstance(stmt.target, ast.Name)
                    and isinstance(stmt.value, ast.Call)
                ):
                    call = stmt.value
                    if (
                        isinstance(call.func, ast.Attribute)
                        and call.func.attr == "mapped_column"
                    ):
                        kw = _kwargs_of(call)
                        type_node = call.args[0] if call.args else None
                        type_str = _short_type(type_node) if type_node else "?"
                        nullable = kw.get("nullable")
                        nullable_str = ""
                        if isinstance(nullable, ast.Constant):
                            nullable_str = "" if nullable.value else " NOT NULL"
                        cols.append({
                            "name": stmt.target.id,
                            "type": type_str,
                            "nullable": nullable_str,
                        })
            if tablename is None:
                continue
            models.append({
                "file": py.name,
                "class": node.name,
                "tablename": tablename,
                "cols": cols,
            })
    models.sort(key=lambda m: m["tablename"])
    return models


def render(models: list[dict], today: str) -> str:
    lines = [
        "# Reference: DB Schema (All SQLAlchemy Models)",
        "",
        "<!-- AUTO-GENERATED. DO NOT EDIT. Regenerate via ./scripts/gen-db-schema.py -->",
        "",
        "> **Status:** ✅ Auto-generated",
        "> **Doc owner:** Platform team",
        f"> **Source of truth:** `backend/app/db/models/`",
        f"> **Last regenerated:** {today}",
        f"> **Total model files:** {len({m['file'] for m in models})}",
        f"> **Total model classes:** {len(models)}",
        f"> **Total tables:** {len(models)}",
        "",
        "---",
        "",
        "## Purpose",
        "",
        "Canonical inventory of every SQLAlchemy model. For per-feature data",
        "semantics, see `docs/features/<feature>.md`.",
        "",
        "## Conventions",
        "",
        "- Every table has a UUID PK (via `UUIDPrimaryKeyMixin`).",
        "- Every table has `created_at` + `updated_at`.",
        "- Tenant-scoped tables extend `TenantScopedModel` (adds",
        "  `tenant_id` + `project_id`) and have a composite index.",
        "- Mutable tables extend `SoftDeleteMixin`.",
        "",
        "## Models by table",
        "",
    ]
    by_file: dict[str, list[dict]] = {}
    for m in models:
        by_file.setdefault(m["file"], []).append(m)
    for f in sorted(by_file):
        items = by_file[f]
        lines.append(f"### `{f}` — {len(items)} model(s)")
        lines.append("")
        for m in items:
            lines.append(f"#### `{m['tablename']}` (`{m['class']}`)")
            lines.append("")
            if m["cols"]:
                lines.append("| Column | Type | Nullable |")
                lines.append("|---|---|---|")
                for c in m["cols"]:
                    lines.append(f"| `{c['name']}` | `{c['type']}` | {c['nullable'] or 'NULL'} |")
                lines.append("")
            else:
                lines.append("_(no mapped_column statements parsed)_")
                lines.append("")
    lines.append("---")
    lines.append("")
    lines.append(f"_Generated by `scripts/gen-db-schema.py` on {today}._")
    lines.append("")
    return "\n".join(lines)


def main() -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("--check", action="store_true")
    ap.add_argument("--dry-run", action="store_true")
    args = ap.parse_args()

    models = collect_models()
    today = dt.date.today().isoformat()
    body = render(models, today)

    if args.dry_run:
        sys.stdout.write(body)
        return 0

    if args.check:
        if not OUT.exists():
            print(f"::error::{OUT} missing — run ./scripts/gen-db-schema.py", file=sys.stderr)
            return 1
        existing = OUT.read_text(encoding="utf-8")
        if existing != body:
            print(f"::error::{OUT} is stale. Run ./scripts/gen-db-schema.py", file=sys.stderr)
            return 1
        return 0

    OUT.write_text(body, encoding="utf-8")
    print(f"wrote {OUT} ({len(models)} models, {len({m['file'] for m in models})} files)")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

```python
#!/usr/bin/env python3
"""Verify every docs/goals/step-*.md has a top-of-file Status header.

The header must appear in the first 10 lines and match one of:
  > **Status:** <state>
  **Status:** <state>

where <state> resolves (via a small synonym table) to one of:
  implemented | in-progress | cancelled

For 'implemented' goals, a 'Last verified:' date in the first 30 lines is
also required.

Usage:
    ./scripts/check-goal-status.sh            # CI mode
    ./scripts/check-goal-status.sh --list     # show every goal + resolved status

Exit codes:
    0  every goal has a valid Status header
    1  one or more goals are missing / wrong
"""
from __future__ import annotations

import argparse
import datetime as dt
import re
import sys
from pathlib import Path

REPO = Path(__file__).resolve().parent.parent
GOALS = REPO / "docs" / "goals"

STATES = {"implemented", "in-progress", "cancelled"}
SYNONYMS = {
    # Prose → canonical
    "ready to run": "in-progress",
    "in progress": "in-progress",
    "in-progress": "in-progress",
    "draft": "in-progress",
    "tested": "in-progress",
    "production": "implemented",
    "shipped": "implemented",
    "shipped.": "implemented",
    "complete": "implemented",
    "✅ complete": "implemented",
    "✅ complete — 10/10 zones implemented and verified.": "implemented",
    "goal met": "implemented",
    "outdated": "cancelled",
    "out of scope": "cancelled",
    "cancelled": "cancelled",
    "canceled": "cancelled",
}


STATUS_RE = re.compile(
    r"""^(?:>\s*)?\*\*Status:\*\*\s+(?P<raw>.+?)\s*$""",
    re.IGNORECASE | re.MULTILINE,
)
LAST_VERIFIED_RE = re.compile(
    r"""^(?:>\s*)?\*\*Last verified:\*\*\s+(?P<date>\d{4}-\d{2}-\d{2})\s*$""",
    re.IGNORECASE | re.MULTILINE,
)


def resolve(raw: str) -> str | None:
    raw = raw.strip().rstrip(".").lower()
    if raw in STATES:
        return raw
    return SYNONYMS.get(raw)


def check_one(path: Path) -> tuple[str, list[str]]:
    """Return (resolved_status, problems). problems is empty on success."""
    text = path.read_text(encoding="utf-8")
    head = "\n".join(text.splitlines()[:10])
    m = STATUS_RE.search(head)
    if not m:
        return ("?", [f"missing top-of-file **Status:** header in {path}"])
    state = resolve(m.group("raw"))
    if state is None:
        return ("?", [f"unknown Status value '{m.group('raw').strip()}' in {path}"])
    problems: list[str] = []
    if state == "implemented":
        if not LAST_VERIFIED_RE.search("\n".join(text.splitlines()[:30])):
            problems.append(
                f"{path} is 'implemented' but lacks '**Last verified:** YYYY-MM-DD' in first 30 lines"
            )
    return (state, problems)


def main() -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("--list", action="store_true")
    args = ap.parse_args()

    step_files = sorted(GOALS.glob("step-*.md"))
    # Exclude non-primary: deliverable, verification, rationale, v2, v3, v4.
    primaries = [
        p for p in step_files
        if not re.search(r"-(deliverable|verification|rationale|v\d)\.md$", p.name)
    ]
    bad: list[str] = []
    for p in primaries:
        state, problems = check_one(p)
        if args.list:
            print(f"{state}\t{p.relative_to(REPO)}")
        bad.extend(problems)
    if bad:
        print("\n".join(f"::error::{p}" for p in bad), file=sys.stderr)
        return 1
    if not args.list:
        print(f"✅ All {len(primaries)} primary goal docs have a valid Status header.")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

```python
Usage:
    ./scripts/check-goal-reality.sh
"""

import re
import subprocess
import sys
from pathlib import Path

REPO = Path(__file__).resolve().parent.parent
GOALS = REPO / "docs" / "goals"
V1 = REPO / "backend" / "app" / "api" / "v1"

# Lines like "- Files: `backend/app/foo.py`, `apps/forge/lib/x.ts`"
TARGETS_RE = re.compile(r"(?:Files|Targets|Routes|Models|Endpoints|Paths):\s*`([^`]+)`", re.IGNORECASE)

def all_routes() -> set[str]:
    out: set[str] = set()
    import ast
    for py in V1.rglob("*.py"):
        if py.name in {"__init__.py", "router.py", "_package_wiring.py"}:
            continue

                if call.func.attr not in {"get", "post", "put", "patch", "delete"}:
                    continue
                if call.args and isinstance(call.args[0], ast.Constant) and isinstance(call.args[0].value, str):
                    out.add(f"/api/v1{call.args[0].value}" if not call.args[0].value.startswith("/api") else call.args[0].value)
    return out


def all_files() -> set[str]:
    out: set[str] = set()
    for p in REPO.rglob("*"):
        if not p.is_file():
            continue
        rel = p.relative_to(REPO).as_posix()
        # Only index source-ish files; skip node_modules / .venv / dist.
        if any(seg in rel for seg in ("node_modules/", "/.venv/", "/dist/", "/.astro/", "/__pycache__/", "/.next/")):
            continue
        out.add(rel)
    return out


def status_of(path: Path) -> str:
    text = path.read_text(encoding="utf-8")
    head = "\n".join(text.splitlines()[:10])
    m = re.search(r"\*\*Status:\*\*\s+(?P<raw>.+)", head, re.IGNORECASE)
    if not m:
        return "?"
    raw = m.group("raw").strip().rstrip(".").lower()
    if raw in {"implemented", "in-progress", "cancelled"}:
        return raw
    return {
        "ready to run": "in-progress",
        "shipped": "implemented",
        "complete": "implemented",
        "✅ complete": "implemented",
        "goal met": "implemented",
    }.get(raw, "?")


def main() -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("--verbose", action="store_true")
    args = ap.parse_args()

    files = all_files()
    routes = all_routes()
    problems: list[str] = []
    checked = 0

    for p in sorted(GOALS.glob("step-*.md")):
        if re.search(r"-(deliverable|verification|rationale|v\d)\.md$", p.name):
            continue
        state = status_of(p)
        if state != "implemented":
            continue
        checked += 1
        text = p.read_text(encoding="utf-8")
        targets = TARGETS_RE.findall(text)
        if not targets:
            problems.append(f"{p}: 'implemented' but no Files/Targets/Routes/Models/Endpoints/Paths line")
            continue
        missing = [t for t in targets if t not in files and t not in routes]
        if missing:
            problems.append(
                f"{p}: 'implemented' but targets missing in code: {', '.join(missing)}"
            )
            continue
        # PR or test fallback.
        has_pr = bool(PR_RE.search(text))
        slug = SLUG_RE.search(p.name)
        if not has_pr and slug:
            slug_id = slug.group(1)
            # grep for step-69 style references in tests/ and apps/forge/tests/
            try:
                r = subprocess.run(
                    ["grep", "-rl", f"step-{slug_id}", str(REPO / "backend" / "tests"),
                     str(REPO / "apps" / "forge" / "tests")],
                    capture_output=True, text=True, check=False,
                )
                if not r.stdout.strip():
                    problems.append(
                        f"{p}: 'implemented' has no PR reference and no test file references step-{slug_id}"
                    )
            except FileNotFoundError:
                pass

    if args.verbose:
        print(f"checked {checked} implemented goal docs")
    if problems:
        print("\n".join(f"::error::{p}" for p in problems), file=sys.stderr)
        return 1
    print(f"✅ All {checked} 'implemented' goal docs reference real code paths.")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

```python
#!/usr/bin/env python3
"""Verify docs/plan/README.md checklist items link bidirectionally to phase docs.

Rules:
  1. Every numbered row in the 'Definition of 10/10' table in
     docs/plan/README.md must link to a phase doc (./phase-N.md).
  2. Every phase doc (phase-N.md, N in 1..8) must have a section
     'Checklist items owned' that lists the row numbers it owns.

Usage:
    ./scripts/check-plan-links.sh
"""
from __future__ import annotations

import argparse
import re
import sys
from pathlib import Path

REPO = Path(__file__).resolve().parent.parent
PLAN = REPO / "docs" / "plan"


def parse_master_rows() -> list[tuple[str, str, str]]:
    """Return [(row_num, property, owner_phase)] from the master checklist."""
    text = (PLAN / "README.md").read_text(encoding="utf-8")
    rows: list[tuple[str, str, str]] = []
    # Lines like: "| 1 | ... | 1 |"  (row, property, owner_phase)
    row_re = re.compile(r"^\|\s*(\d+)\s*\|\s*([^|]+?)\s*\|\s*(\d+)\s*\|\s*$", re.MULTILINE)
    for m in row_re.finditer(text):
        n, prop, owner = m.group(1), m.group(2).strip(), m.group(3).strip()
        # Skip header rows where 'owner' is not a digit.
        if not owner.isdigit():
            continue
        rows.append((n, prop, owner))
    return rows


def phase_doc_owns(phase: int, row_num: str) -> bool:
    """Check phase-N.md has 'Checklist items owned' section listing row_num."""
    p = PLAN / f"phase-{phase}.md"
    if not p.exists():
        return False
    text = p.read_text(encoding="utf-8")
    m = re.search(
        r"^##\s*Checklist items owned\s*$([\s\S]*?)(?=^##\s|\Z)",
        text, re.MULTILINE | re.IGNORECASE,
    )
    if not m:
        return False
    section = m.group(1)
    return bool(re.search(rf"\b{re.escape(row_num)}\b", section))


def main() -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("--verbose", action="store_true")
    args = ap.parse_args()

    rows = parse_master_rows()
    problems: list[str] = []
    for n, prop, owner in rows:
        if not phase_doc_owns(int(owner), n):
            problems.append(
                f"row #{n} ('{prop}') owned by phase {owner}, but phase-{owner}.md "
                f"does not list #{n} under 'Checklist items owned'."
            )
    # Reverse direction: every 'Checklist items owned' section must reference
    # an existing master row.
    all_owners = {int(o) for _, _, o in rows}
    all_row_nums = {n for n, _, _ in rows}
    for phase in all_owners:
        p = PLAN / f"phase-{phase}.md"
        text = p.read_text(encoding="utf-8")
        m = re.search(
            r"^##\s*Checklist items owned\s*$([\s\S]*?)(?=^##\s|\Z)",
            text, re.MULTILINE | re.IGNORECASE,
        )
        if not m:
            continue  # already reported above
        section = m.group(1)
        for ref in re.findall(r"\b(\d+)\b", section):
            if ref not in all_row_nums:
                problems.append(
                    f"phase-{phase}.md 'Checklist items owned' references #{ref} which is not in the master checklist."
                )

    if args.verbose:
        print(f"checked {len(rows)} master rows across {len(all_owners)} phase docs")
    if problems:
        print("\n".join(f"::error::{p}" for p in problems), file=sys.stderr)
        return 1
    print(f"✅ Master checklist is bidirectionally linked to phase docs.")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```

```python
#!/usr/bin/env python3
"""Verify the §2 endpoint matrix in
docs/litellm/forge-litellm-integration.md does not name a /api/v1/ path
that doesn't exist in backend/app/api/v1/.

§2 is a Forge-feature → LiteLLM-endpoint matrix. Column 3 is the Forge
Backend call. Some entries are LiteLLM passthroughs (column 4) — those
are exempt.

Rule: any path in column 3 starting with '/api/v1/' MUST exist as a
router path in backend/app/api/v1/. Paths NOT starting with '/api/v1/'
are treated as LiteLLM passthroughs and skipped.

Usage:
    ./scripts/check-litellm-matrix.sh
"""
from __future__ import annotations

import argparse
import ast
import re
import sys
from pathlib import Path

REPO = Path(__file__).resolve().parent.parent
DOC = REPO / "docs" / "litellm" / "forge-litellm-integration.md"
V1 = REPO / "backend" / "app" / "api" / "v1"

# Path fragments in backticks in column 3 of §2 tables. §2 starts at "## 2."
# and ends at "## 3." (or EOF).
SEC2_HEADER = re.compile(r"^##\s*2\.\s", re.MULTILINE)
SEC3_HEADER = re.compile(r"^##\s*3\.\s", re.MULTILINE)
# Table-row with 3 or 4 columns. Capture column 3.
ROW_RE = re.compile(r"^\|\s*[^|]+\s*\|\s*[^|]+\s*\|\s*(?P<c3>[^|]+?)\s*(?:\|\s*[^|]*\s*\|)?\s*$")
PATH_RE = re.compile(r"`(/[a-zA-Z_][a-zA-Z0-9_/{}\-]*)`")


def all_routes() -> set[str]:
    out: set[str] = set()
    for py in V1.rglob("*.py"):
        if py.name in {"__init__.py", "router.py", "_package_wiring.py"}:
            continue
        try:
            tree = ast.parse(py.read_text(encoding="utf-8"))
        except SyntaxError:
            continue
        for node in ast.walk(tree):
            if not isinstance(node, (ast.FunctionDef, ast.AsyncFunctionDef)):
                continue
            for dec in node.decorator_list:
                call = dec if isinstance(dec, ast.Call) else None
                if call is None or not isinstance(call.func, ast.Attribute):
                    continue
                if call.func.attr not in {"get", "post", "put", "patch", "delete"}:
                    continue
                if call.args and isinstance(call.args[0], ast.Constant) and isinstance(call.args[0].value, str):
                    p = call.args[0].value
                    if p.startswith("/"):
                        out.add(f"/api/v1{p}" if not p.startswith("/api") else p)
    return out


def main() -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("--verbose", action="store_true")
    args = ap.parse_args()

    text = DOC.read_text(encoding="utf-8")
    sec2 = SEC2_HEADER.search(text)
    sec3 = SEC3_HEADER.search(text)
    if not sec2:
        print(f"::error::cannot find '## 2.' section in {DOC}", file=sys.stderr)
        return 1
    body = text[sec2.end():sec3.start() if sec3 else None]

    routes = all_routes()
    problems: list[str] = []
    checked = 0

    for line in body.splitlines():
        if not line.startswith("|"):
            continue
        m = ROW_RE.match(line)
        if not m:
            continue
        c3 = m.group("c3")
        for path_match in PATH_RE.finditer(c3):
            path = path_match.group(1)
            # Only check /api/v1/ paths; LiteLLM passthroughs are exempt.
            if not path.startswith("/api/v1/"):
                continue
            # Allow path templates with {id} — strip for comparison.
            stripped = re.sub(r"\{[^}]+\}", "{id}", path)
            checked += 1
            # Match either the literal or the stripped form.
            if path in routes or stripped in routes:
                continue
            problems.append(f"{DOC}: §2 mentions '{path}' but no router defines it")

    if args.verbose:
        print(f"checked {checked} /api/v1/ path mentions in §2")
    if problems:
        print("\n".join(f"::error::{p}" for p in problems), file=sys.stderr)
        return 1
    print(f"✅ §2 endpoint matrix references {checked} /api/v1/ paths, all real.")
    return 0


if __name__ == "__main__":
    sys.exit(main())
```
